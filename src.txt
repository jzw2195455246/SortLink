<property value="select 1" name="validationQuery"/>

<!-- 在将池中连接返回给调用者之前，用于验证这些连接的 SQL 查询。如果指定该值，则该查询不必返回任何数据，只是不抛出SQLException 异常。默认为 null -->


<property value="30000" name="validationInterval"/>

<!-- 为避免过度验证而设定的频率时间值（以秒计）。最多以这种频率运行验证。如果连接应该进行验证，但却没能在此间隔时间内得到验证，则会重新对其进行验证。默认为30000（30 秒） -->


</bean>
